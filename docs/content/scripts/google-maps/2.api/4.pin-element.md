---
title: <ScriptGoogleMapsPinElement>
---

Customizable pin marker. Place as a child of [`<ScriptGoogleMapsAdvancedMarkerElement>`{lang="html"}](/scripts/google-maps/api/advanced-marker-element).

::script-types{script-key="google-maps" filter="ScriptGoogleMapsPinElement"}
::

## Usage

```vue
<template>
  <ScriptGoogleMaps api-key="your-api-key">
    <ScriptGoogleMapsAdvancedMarkerElement
      :position="{ lat: -34.407, lng: 150.654 }"
    >
      <ScriptGoogleMapsPinElement
        :options="{
          scale: 1.5,
          background: '#FF0000',
          borderColor: '#CC0000',
          glyphColor: '#FFFFFF',
        }"
      />
    </ScriptGoogleMapsAdvancedMarkerElement>
  </ScriptGoogleMaps>
</template>
```

### Text Glyph

Display a character or short label inside the pin.

```vue
<template>
  <ScriptGoogleMaps api-key="your-api-key">
    <ScriptGoogleMapsAdvancedMarkerElement
      :position="{ lat: -34.397, lng: 150.644 }"
    >
      <ScriptGoogleMapsPinElement
        :options="{
          glyph: 'A',
          glyphColor: '#FFFFFF',
          background: '#4285F4',
          borderColor: '#1a73e8',
        }"
      />
    </ScriptGoogleMapsAdvancedMarkerElement>
  </ScriptGoogleMaps>
</template>
```

### Hide the Glyph

Set `glyph` to an empty string for a solid colored pin without any icon or text.

```vue
<template>
  <ScriptGoogleMaps api-key="your-api-key">
    <ScriptGoogleMapsAdvancedMarkerElement
      :position="{ lat: -34.397, lng: 150.644 }"
    >
      <ScriptGoogleMapsPinElement
        :options="{
          glyph: '',
          background: '#FBBC04',
          scale: 1.2,
        }"
      />
    </ScriptGoogleMapsAdvancedMarkerElement>
  </ScriptGoogleMaps>
</template>
```

### Multiple Pin Styles

Combine different pin configurations to distinguish marker categories.

```vue
<script setup lang="ts">
const restaurants = [{ lat: -34.397, lng: 150.644 }]
const hotels = [{ lat: -34.407, lng: 150.654 }]
</script>

<template>
  <ScriptGoogleMaps api-key="your-api-key">
    <ScriptGoogleMapsAdvancedMarkerElement
      v-for="(pos, i) in restaurants"
      :key="`r-${i}`"
      :position="pos"
    >
      <ScriptGoogleMapsPinElement :options="{ background: '#EA4335', glyphColor: '#fff', glyph: '🍕' }" />
    </ScriptGoogleMapsAdvancedMarkerElement>

    <ScriptGoogleMapsAdvancedMarkerElement
      v-for="(pos, i) in hotels"
      :key="`h-${i}`"
      :position="pos"
    >
      <ScriptGoogleMapsPinElement :options="{ background: '#4285F4', glyphColor: '#fff', glyph: '🏨' }" />
    </ScriptGoogleMapsAdvancedMarkerElement>
  </ScriptGoogleMaps>
</template>
```

::callout
The `options` prop is reactive. Changing it updates the pin appearance in place. If you need fully custom HTML instead of the pin shape, use the `#content` slot on `AdvancedMarkerElement` instead.
::
